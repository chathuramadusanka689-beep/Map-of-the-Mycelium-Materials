    <script>
        // Application State
        const app = {
            data: [],
            filteredData: [],
            currentPage: 1,
            rowsPerPage: 10,
            sortColumn: null,
            sortDirection: 'asc',
            searchTerm: '',
            selectedRows: new Set(),
            currentRecord: null,
            originalFile: null,
            
            // Initialize
            init() {
                this.loadFromStorage();
                this.setupDragAndDrop();
                this.setupTheme();
                
                // Check for saved data
                if (this.data.length > 0) {
                    this.showDataView();
                    this.updateStats();
                    this.populateFilters();
                    this.renderTable();
                }
            },

            // Theme Management
            setupTheme() {
                const savedTheme = localStorage.getItem('theme') || 'light';
                document.documentElement.setAttribute('data-theme', savedTheme);
                if (savedTheme === 'dark') {
                    document.documentElement.classList.add('dark');
                }
            },

            toggleTheme() {
                const html = document.documentElement;
                const currentTheme = html.getAttribute('data-theme');
                const newTheme = currentTheme === 'light' ? 'dark' : 'light';
                
                html.setAttribute('data-theme', newTheme);
                localStorage.setItem('theme', newTheme);
                
                if (newTheme === 'dark') {
                    html.classList.add('dark');
                } else {
                    html.classList.remove('dark');
                }
            },

            // Drag and Drop
            setupDragAndDrop() {
                const zone = document.getElementById('uploadZone');
                
                ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
                    zone.addEventListener(eventName, (e) => {
                        e.preventDefault();
                        e.stopPropagation();
                    });
                });

                ['dragenter', 'dragover'].forEach(eventName => {
                    zone.addEventListener(eventName, () => {
                        zone.classList.add('dragover');
                    });
                });

                ['dragleave', 'drop'].forEach(eventName => {
                    zone.addEventListener(eventName, () => {
                        zone.classList.remove('dragover');
                    });
                });

                zone.addEventListener('drop', (e) => {
                    const files = e.dataTransfer.files;
                    if (files.length) {
                        this.processFile(files[0]);
                    }
                });

                zone.addEventListener('click', () => {
                    document.getElementById('fileInput').click();
                });
            },

            // File Upload
            handleFileUpload(event) {
                const file = event.target.files[0];
                if (file) {
                    this.processFile(file);
                }
            },

            processFile(file) {
                // Store original file for download
                this.originalFile = file;
                
                const reader = new FileReader();
                reader.onload = (e) => {
                    try {
                        const data = new Uint8Array(e.target.result);
                        const workbook = XLSX.read(data, { type: 'array' });
                        const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
                        const jsonData = XLSX.utils.sheet_to_json(firstSheet, { header: 1 });
                        
                        if (jsonData.length < 2) {
                            throw new Error('File appears to be empty or missing headers');
                        }
                        
                        this.parseData(jsonData);
                        this.showToast('File uploaded successfully!', 'success');
                    } catch (error) {
                        this.showToast('Error parsing file: ' + error.message, 'error');
                    }
                };
                reader.readAsArrayBuffer(file);
            },

            parseData(rawData) {
                const headers = rawData[0].map(h => h.toString().toLowerCase().trim());
                const rows = rawData.slice(1);
                
                this.data = rows.map((row, index) => {
                    const obj = { id: index + 1, selected: false };
                    headers.forEach((header, i) => {
                        const key = this.normalizeHeader(header);
                        obj[key] = row[i] || '';
                    });
                    return obj;
                }).filter(row => row.title || row.paperTitle); // Filter out empty rows
                
                // Normalize field names
                this.data = this.data.map(row => ({
                    id: row.id,
                    title: row.title || row.papertitle || row.paper_title || 'Untitled',
                    authors: row.authors || row.author || 'Unknown',
                    year: parseInt(row.year) || new Date().getFullYear(),
                    materialType: row.materialtype || row.material_type || row.material || 'Unspecified',
                    properties: row.properties || row.mechanical || row.thermal || '',
                    source: row.source || row.doi || row.link || '',
                    notes: row.notes || row.comments || '',
                    keywords: row.keywords || ''
                }));
                
                this.filteredData = [...this.data];
                this.saveToStorage();
                this.showDataView();
                this.updateStats();
                this.populateFilters();
                this.renderTable();
            },

            normalizeHeader(header) {
                return header.replace(/[^a-zA-Z0-9]/g, '').toLowerCase();
            },

            // Storage
            saveToStorage() {
                localStorage.setItem('myceliumData', JSON.stringify(this.data));
                if (this.originalFile) {
                    // Store file metadata since we can't store the file itself
                    localStorage.setItem('originalFileName', this.originalFile.name);
                }
            },

            loadFromStorage() {
                const saved = localStorage.getItem('myceliumData');
                if (saved) {
                    try {
                        this.data = JSON.parse(saved);
                        this.filteredData = [...this.data];
                    } catch (e) {
                        console.error('Error loading saved data:', e);
                    }
                }
            },

            // UI Updates
            showDataView() {
                document.getElementById('uploadZone').classList.add('hidden');
                document.getElementById('controlsSection').classList.remove('hidden');
                document.getElementById('tableSection').classList.remove('hidden');
            },

            updateStats() {
                document.getElementById('totalRecords').textContent = this.data.length;
                
                // Unique authors
                const authors = new Set();
                this.data.forEach(row => {
                    if (row.authors) {
                        row.authors.split(/[;,]/).forEach(a => authors.add(a.trim()));
                    }
                });
                document.getElementById('uniqueAuthors').textContent = authors.size;
                
                // Year range
                const years = this.data.map(r => r.year).filter(y => y);
                if (years.length) {
                    const min = Math.min(...years);
                    const max = Math.max(...years);
                    document.getElementById('yearRange').textContent = min === max ? min : `${min}-${max}`;
                }
            },

            populateFilters() {
                // Year filter
                const years = [...new Set(this.data.map(r => r.year))].filter(y => y).sort((a, b) => b - a);
                const yearSelect = document.getElementById('yearFilter');
                yearSelect.innerHTML = '<option value="">All Years</option>' + 
                    years.map(y => `<option value="${y}">${y}</option>`).join('');
                
                // Material filter
                const materials = [...new Set(this.data.map(r => r.materialType))].filter(m => m).sort();
                const materialSelect = document.getElementById('materialFilter');
                materialSelect.innerHTML = '<option value="">All Materials</option>' + 
                    materials.map(m => `<option value="${m}">${m}</option>`).join('');
            },

            // Search and Filter
            handleSearch(term) {
                this.searchTerm = term.toLowerCase();
                this.applyFilters();
            },

            applyFilters() {
                const yearFilter = document.getElementById('yearFilter').value;
                const materialFilter = document.getElementById('materialFilter').value;
                
                this.filteredData = this.data.filter(row => {
                    // Search term
                    if (this.searchTerm) {
                        const searchable = `${row.title} ${row.authors} ${row.keywords} ${row.notes}`.toLowerCase();
                        if (!searchable.includes(this.searchTerm)) return false;
                    }
                    
                    // Year filter
                    if (yearFilter && row.year.toString() !== yearFilter) return false;
                    
                    // Material filter
                    if (materialFilter && row.materialType !== materialFilter) return false;
                    
                    return true;
                });
                
                this.currentPage = 1;
                this.renderTable();
                this.updateResultCount();
            },

            resetFilters() {
                document.getElementById('searchInput').value = '';
                document.getElementById('yearFilter').value = '';
                document.getElementById('materialFilter').value = '';
                this.searchTerm = '';
                this.filteredData = [...this.data];
                this.currentPage = 1;
                this.renderTable();
                this.updateResultCount();
            },

            updateResultCount() {
                document.getElementById('showingCount').textContent = this.filteredData.length;
                document.getElementById('totalCount').textContent = this.data.length;
                
                const emptyState = document.getElementById('emptyState');
                const tableSection = document.getElementById('tableSection');
                
                if (this.filteredData.length === 0) {
                    emptyState.classList.remove('hidden');
                    tableSection.classList.add('hidden');
                } else {
                    emptyState.classList.add('hidden');
                    tableSection.classList.remove('hidden');
                }
            },

            // Sorting
            sort(column) {
                if (this.sortColumn === column) {
                    this.sortDirection = this.sortDirection === 'asc' ? 'desc' : 'asc';
                } else {
                    this.sortColumn = column;
                    this.sortDirection = 'asc';
                }
                
                // Update UI indicators
                document.querySelectorAll('th').forEach(th => {
                    th.classList.remove('sort-asc', 'sort-desc');
                });
                
                const th = document.querySelector(`th[onclick="app.sort('${column}')"]`);
                if (th) {
                    th.classList.add(this.sortDirection === 'asc' ? 'sort-asc' : 'sort-desc');
                }
                
                // Sort data
                this.filteredData.sort((a, b) => {
                    let valA = a[column] || '';
                    let valB = b[column] || '';
                    
                    if (typeof valA === 'string') valA = valA.toLowerCase();
                    if (typeof valB === 'string') valB = valB.toLowerCase();
                    
                    if (valA < valB) return this.sortDirection === 'asc' ? -1 : 1;
                    if (valA > valB) return this.sortDirection === 'asc' ? 1 : -1;
                    return 0;
                });
                
                this.renderTable();
            },

            // Pagination
            changeRowsPerPage(value) {
                this.rowsPerPage = parseInt(value);
                this.currentPage = 1;
                this.renderTable();
            },

            prevPage() {
                if (this.currentPage > 1) {
                    this.currentPage--;
                    this.renderTable();
                }
            },

            nextPage() {
                const maxPage = Math.ceil(this.filteredData.length / this.rowsPerPage);
                if (this.currentPage < maxPage) {
                    this.currentPage++;
                    this.renderTable();
                }
            },

            // Table Rendering
            renderTable() {
                const tbody = document.getElementById('tableBody');
                const start = (this.currentPage - 1) * this.rowsPerPage;
                const end = start + this.rowsPerPage;
                const pageData = this.filteredData.slice(start, end);
                
                tbody.innerHTML = pageData.map(row => {
                    const titleHighlighted = this.highlightText(row.title, this.searchTerm);
                    const authorsHighlighted = this.highlightText(row.authors, this.searchTerm);
                    
                    return `
                        <tr class="table-row border-b border-gray-100 dark:border-gray-700 hover:bg-gray-50 dark:hover:bg-slate-800/50 animate-fade-in">
                            <td class="px-6 py-4">
                                <input type="checkbox" class="custom-checkbox" 
                                    ${this.selectedRows.has(row.id) ? 'checked' : ''} 
                                    onchange="app.toggleRowSelection(${row.id})">
                            </td>
                            <td class="px-6 py-4">
                                <div class="font-medium text-gray-900 dark:text-gray-100">${titleHighlighted}</div>
                                ${row.keywords ? `<div class="text-xs text-gray-500 mt-1 flex gap-1 flex-wrap">${row.keywords.split(/[,;]/).map(k => `<span class="bg-gray-100 dark:bg-slate-700 px-2 py-0.5 rounded">${k.trim()}</span>`).join('')}</div>` : ''}
                            </td>
                            <td class="px-6 py-4 text-sm text-gray-600 dark:text-gray-400">${authorsHighlighted}</td>
                            <td class="px-6 py-4">
                                <span class="inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium bg-mycelium-100 text-mycelium-800 dark:bg-mycelium-900/30 dark:text-mycelium-300">
                                    ${row.year}
                                </span>
                            </td>
                            <td class="px-6 py-4 text-sm text-gray-600 dark:text-gray-400">${row.materialType}</td>
                            <td class="px-6 py-4">
                                <button onclick="app.viewDetails(${row.id})" class="text-mycelium-600 hover:text-mycelium-700 font-medium text-sm flex items-center gap-1">
                                    View <i class="fas fa-arrow-right text-xs"></i>
                                </button>
                            </td>
                        </tr>
                    `;
                }).join('');
                
                // Update pagination controls
                const totalPages = Math.ceil(this.filteredData.length / this.rowsPerPage) || 1;
                document.getElementById('currentPage').textContent = this.currentPage;
                document.getElementById('totalPages').textContent = totalPages;
                document.getElementById('prevBtn').disabled = this.currentPage === 1;
                document.getElementById('nextBtn').disabled = this.currentPage >= totalPages;
                
                // Update select all checkbox
                const selectAll = document.getElementById('selectAll');
                const visibleIds = pageData.map(r => r.id);
                const allSelected = visibleIds.length > 0 && visibleIds.every(id => this.selectedRows.has(id));
                selectAll.checked = allSelected;
                
                this.updateResultCount();
            },

            highlightText(text, term) {
                if (!term || !text) return text;
                const regex = new RegExp(`(${term})`, 'gi');
                return text.replace(regex, '<span class="highlight-match">$1</span>');
            },

            // Selection
            toggleRowSelection(id) {
                if (this.selectedRows.has(id)) {
                    this.selectedRows.delete(id);
                } else {
                    this.selectedRows.add(id);
                }
                this.renderTable();
            },

            toggleSelectAll() {
                const start = (this.currentPage - 1) * this.rowsPerPage;
                const end = start + this.rowsPerPage;
                const pageIds = this.filteredData.slice(start, end).map(r => r.id);
                const allSelected = pageIds.every(id => this.selectedRows.has(id));
                
                if (allSelected) {
                    pageIds.forEach(id => this.selectedRows.delete(id));
                } else {
                    pageIds.forEach(id => this.selectedRows.add(id));
                }
                
                this.renderTable();
            },

            // Modal
            viewDetails(id) {
                const record = this.data.find(r => r.id === id);
                if (!record) return;
                
                this.currentRecord = record;
                const modal = document.getElementById('detailModal');
                const content = document.getElementById('modalContent');
                
                content.innerHTML = `
                    <div class="space-y-6">
                        <div>
                            <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Title</label>
                            <h3 class="text-lg font-semibold text-gray-900 dark:text-gray-100 mt-1">${record.title}</h3>
                        </div>
                        
                        <div class="grid grid-cols-1 md:grid-cols-2 gap-6">
                            <div>
                                <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Authors</label>
                                <p class="text-gray-700 dark:text-gray-300 mt-1">${record.authors}</p>
                            </div>
                            <div>
                                <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Year</label>
                                <p class="text-gray-700 dark:text-gray-300 mt-1">${record.year}</p>
                            </div>
                        </div>
                        
                        <div>
                            <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Material Type</label>
                            <p class="text-gray-700 dark:text-gray-300 mt-1">${record.materialType}</p>
                        </div>
                        
                        ${record.properties ? `
                        <div>
                            <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Properties</label>
                            <div class="mt-2 p-4 bg-gray-50 dark:bg-slate-900/50 rounded-lg">
                                <p class="text-gray-700 dark:text-gray-300 whitespace-pre-wrap">${record.properties}</p>
                            </div>
                        </div>
                        ` : ''}
                        
                        ${record.source ? `
                        <div>
                            <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Source / DOI</label>
                            <a href="${record.source.startsWith('http') ? record.source : 'https://doi.org/' + record.source}" 
                               target="_blank" 
                               class="text-mycelium-600 hover:text-mycelium-700 mt-1 block break-all">
                                ${record.source} <i class="fas fa-external-link-alt text-xs ml-1"></i>
                            </a>
                        </div>
                        ` : ''}
                        
                        ${record.notes ? `
                        <div>
                            <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Notes</label>
                            <div class="mt-2 p-4 bg-yellow-50 dark:bg-yellow-900/20 border-l-4 border-yellow-400 rounded-r-lg">
                                <p class="text-gray-700 dark:text-gray-300 whitespace-pre-wrap">${record.notes}</p>
                            </div>
                        </div>
                        ` : ''}
                        
                        ${record.keywords ? `
                        <div>
                            <label class="text-xs font-semibold text-gray-500 dark:text-gray-400 uppercase tracking-wider">Keywords</label>
                            <div class="flex gap-2 mt-2 flex-wrap">
                                ${record.keywords.split(/[,;]/).map(k => `
                                    <span class="px-3 py-1 bg-mycelium-100 dark:bg-mycelium-900/30 text-mycelium-800 dark:text-mycelium-300 rounded-full text-sm">
                                        ${k.trim()}
                                    </span>
                                `).join('')}
                            </div>
                        </div>
                        ` : ''}
                    </div>
                `;
                
                modal.classList.remove('hidden');
                document.body.style.overflow = 'hidden';
            },

            closeModal() {
                document.getElementById('detailModal').classList.add('hidden');
                document.body.style.overflow = '';
                this.currentRecord = null;
            },

            // Downloads
            downloadOriginal() {
                if (this.originalFile) {
                    const url = URL.createObjectURL(this.originalFile);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = this.originalFile.name;
                    a.click();
                    URL.revokeObjectURL(url);
                    this.showToast('Original file downloaded', 'success');
                } else {
                    // If no file stored, export current data as original format
                    this.exportToExcel(this.data, 'mycelium_data_original.xlsx');
                }
            },

            downloadFiltered() {
                const dataToExport = this.selectedRows.size > 0 
                    ? this.data.filter(r => this.selectedRows.has(r.id))
                    : this.filteredData;
                
                if (dataToExport.length === 0) {
                    this.showToast('No data to export', 'error');
                    return;
                }
                
                this.exportToExcel(dataToExport, `mycelium_filtered_${new Date().toISOString().split('T')[0]}.xlsx`);
                this.showToast(`Exported ${dataToExport.length} records`, 'success');
            },

            downloadCurrentRecord() {
                if (!this.currentRecord) return;
                this.exportToExcel([this.currentRecord], `record_${this.currentRecord.id}.xlsx`);
                this.showToast('Record downloaded', 'success');
            },

            exportToExcel(data, filename) {
                const worksheet = XLSX.utils.json_to_sheet(data.map(({id, selected, ...rest}) => rest));
                const workbook = XLSX.utils.book_new();
                XLSX.utils.book_append_sheet(workbook, worksheet, 'Data');
                XLSX.writeFile(workbook, filename);
            },

            // Toast
            showToast(message, type = 'success') {
                const toast = document.getElementById('toast');
                const icon = document.getElementById('toastIcon');
                const msg = document.getElementById('toastMessage');
                
                msg.textContent = message;
                icon.className = type === 'success' 
                    ? 'fas fa-check-circle text-mycelium-400 dark:text-mycelium-600'
                    : 'fas fa-exclamation-circle text-red-400';
                
                toast.classList.remove('translate-y-20', 'opacity-0');
                
                setTimeout(() => {
                    toast.classList.add('translate-y-20', 'opacity-0');
                }, 3000);
            }
        };

        // Initialize app when DOM is ready
        document.addEventListener('DOMContentLoaded', () => {
            app.init();
        });

        // Close modal on escape key
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                app.closeModal();
            }
        });
    </script>
